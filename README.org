* 概要

** システム概要

予算評価レビューを支援するWebアプリケーション。
DS-110・DS-910標準に準拠した調達審査において、評価者とPJMOの往復レビュー（ラリー）を
構造化・履歴管理し、AIによる自動レビューと監査ログ出力を提供する。

** レビューの流れ

#+begin_example
評価者 →[issue] 指摘
   ↓
PJMO   →[reply] 回答
   ↓
評価者 →[review] 再評価
   ↓
評価者 →[close] 確定
#+end_example

#+begin_src python :tangle org_tangle.py
#!/usr/bin/env python3
"""org_tangle.py — Org-babel tangle を Python で再実装する.

org-babel-tangle と同等のロジックで README.org から全ソースファイルを生成する。

対応する Org-babel 機能:
- #+BEGIN_SRC lang :tangle <path>  — ブロックレベルの tangle 指定
- :PROPERTIES: / :header-args:lang: :tangle <path>  — セクションレベルのデフォルト
- :tangle no  — スキップ
- #+PROPERTY: header-args [:mkdirp yes  — ファイルレベルのデフォルト
- 同一ファイルへの複数ブロック追記（org-babel と同じ動作）
- noweb 参照は未対応（retranscribe には不要）

使い方:
    python org_tangle.py                     # カレントの README.org を処理
    python org_tangle.py README.org          # ファイル指定
    python org_tangle.py README.org --dry-run  # ファイル一覧のみ表示
    python org_tangle.py README.org --out /tmp/out  # 出力先を指定
"""

from __future__ import annotations

import argparse
import os
import re
import sys
import textwrap
from dataclasses import dataclass, field
from pathlib import Path
from typing import Optional


# ---------------------------------------------------------------------------
# データ構造
# ---------------------------------------------------------------------------

@dataclass
class Block:
    """1つの #+BEGIN_SRC ブロック."""
    tangle: str           # 出力ファイルパス
    lang: str             # 言語識別子
    content: str          # ブロック本文
    src_line: int         # 元ファイルの行番号（デバッグ用）


# ---------------------------------------------------------------------------
# パーサー
# ---------------------------------------------------------------------------

_BEGIN_SRC_RE = re.compile(
    r"^\s*#\+BEGIN_SRC\s+(\S+)(.*)?$", re.IGNORECASE
)
_END_SRC_RE = re.compile(r"^\s*#\+END_SRC\s*$", re.IGNORECASE)

_HEADER_ARGS_RE = re.compile(
    r"^\s*:header-args(?::(\S+))?:\s+(.+)$", re.IGNORECASE
)

_PROPERTY_RE = re.compile(
    r"^\s*#\+PROPERTY:\s+header-args(?::(\S+))?\s+(.+)$", re.IGNORECASE
)

def _extract_tangle(header_rest: str) -> Optional[str]:
    """ヘッダー文字列から :tangle <value> を抽出する."""
    m = re.search(r":tangle\s+(\S+)", header_rest)
    if m:
        v = m.group(1)
        return None if v.lower() == "no" else v
    return None


def parse_org(text: str) -> list[Block]:
    """Org テキストを走査して tangle 対象ブロックのリストを返す."""

    lines = text.splitlines()
    blocks: list[Block] = []

    file_defaults: dict[str, Optional[str]] = {}
    section_stack: list[dict[str, Optional[str]]] = [{}]
    current_level = 0

    in_properties = False
    in_src = False
    src_lang = ""
    src_tangle: Optional[str] = None
    src_lines: list[str] = []
    src_start_line = 0

    for lineno, line in enumerate(lines, 1):

        if not in_src:
            m = _PROPERTY_RE.match(line)
            if m:
                lang_key = (m.group(1) or "").lower()
                tangle_val = _extract_tangle(m.group(2))
                file_defaults[lang_key] = tangle_val
                continue

        if not in_src:
            h = re.match(r"^(\*+)\s", line)
            if h:
                lvl = len(h.group(1))
                while len(section_stack) > lvl:
                    section_stack.pop()
                while len(section_stack) <= lvl:
                    section_stack.append(dict(section_stack[-1]))
                current_level = lvl
                in_properties = False
                continue

        if not in_src:
            if line.strip() == ":PROPERTIES:":
                in_properties = True
                continue
            if line.strip() == ":END:":
                in_properties = False
                continue
            if in_properties:
                m = _HEADER_ARGS_RE.match(line)
                if m:
                    lang_key = (m.group(1) or "").lower()
                    tangle_val = _extract_tangle(m.group(2))
                    if tangle_val is not None:
                        section_stack[-1][lang_key] = tangle_val
                    elif ":tangle" in m.group(2).lower():
                        section_stack[-1][lang_key] = None
                continue

        if not in_src:
            m = _BEGIN_SRC_RE.match(line)
            if m:
                lang = m.group(1).lower()
                header_rest = m.group(2) or ""

                block_tangle = _extract_tangle(header_rest)

                if block_tangle is None and ":tangle" not in header_rest.lower():
                    sec = section_stack[-1]
                    if lang in sec:
                        block_tangle = sec[lang]
                    elif "" in sec:
                        block_tangle = sec[""]
                    elif lang in file_defaults:
                        block_tangle = file_defaults[lang]
                    elif "" in file_defaults:
                        block_tangle = file_defaults[""]

                in_src = True
                src_lang = lang
                src_tangle = block_tangle
                src_lines = []
                src_start_line = lineno
                continue

        else:
            if _END_SRC_RE.match(line):
                if src_tangle:
                    blocks.append(Block(
                        tangle=src_tangle,
                        lang=src_lang,
                        content="\n".join(src_lines),
                        src_line=src_start_line,
                    ))
                in_src = False
                src_lang = ""
                src_tangle = None
                src_lines = []
                continue
            src_lines.append(line)

    return blocks


# ---------------------------------------------------------------------------
# 書き出し
# ---------------------------------------------------------------------------

def write_files(
    blocks: list[Block],
    base_dir: Path,
    dry_run: bool = False,
) -> dict[str, int]:
    """ブロック群をファイルに書き出す."""

    file_blocks: dict[str, list[Block]] = {}
    for block in blocks:
        file_blocks.setdefault(block.tangle, []).append(block)

    result: dict[str, int] = {}

    for rel_path, blks in file_blocks.items():
        out_path = base_dir / rel_path
        result[rel_path] = len(blks)

        if dry_run:
            continue

        out_path.parent.mkdir(parents=True, exist_ok=True)
        with out_path.open("w", encoding="utf-8") as f:
            for i, blk in enumerate(blks):
                if i > 0:
                    f.write("\n")
                content = textwrap.dedent(blk.content)
                f.write(content)
                if content and not content.endswith("\n"):
                    f.write("\n")

    return result


# ---------------------------------------------------------------------------
# CLI
# ---------------------------------------------------------------------------

def main() -> None:
    parser = argparse.ArgumentParser(
        description="Org-babel tangle を Python で実行する",
        formatter_class=argparse.RawDescriptionHelpFormatter,
        epilog=__doc__,
    )
    parser.add_argument(
        "org_file",
        nargs="?",
        default="README.org",
        help="Org ファイルのパス（デフォルト: README.org）",
    )
    parser.add_argument(
        "--out", "-o",
        default=None,
        help="出力ベースディレクトリ（デフォルト: org ファイルと同じディレクトリ）",
    )
    parser.add_argument(
        "--dry-run", "-n",
        action="store_true",
        help="ファイルを書き出さずにtangleターゲットの一覧のみ表示",
    )
    parser.add_argument(
        "--lang",
        default=None,
        help="指定した言語のブロックのみ処理（例: python, toml）",
    )
    parser.add_argument(
        "--file", "-f",
        default=None,
        help="指定したtangleパスのブロックのみ処理（例: backend/review.py）",
    )
    parser.add_argument(
        "--verbose", "-v",
        action="store_true",
        help="処理中の詳細を表示",
    )

    args = parser.parse_args()

    org_path = Path(args.org_file)
    if not org_path.exists():
        print(f"Error: {org_path} が見つかりません", file=sys.stderr)
        sys.exit(1)

    base_dir = Path(args.out) if args.out else org_path.parent

    text = org_path.read_text(encoding="utf-8")
    blocks = parse_org(text)

    if args.lang:
        blocks = [b for b in blocks if b.lang == args.lang.lower()]
    if args.file:
        blocks = [b for b in blocks if b.tangle == args.file]

    if not blocks:
        print("tangle対象ブロックが見つかりませんでした。")
        sys.exit(0)

    result = write_files(blocks, base_dir, dry_run=args.dry_run)

    if args.dry_run:
        print(f"[dry-run] tangle対象: {len(result)} ファイル")
        for rel, count in sorted(result.items()):
            print(f"  {rel}  ({count} ブロック)")
    else:
        total_blocks = sum(result.values())
        print(f"tangle完了: {len(result)} ファイル / {total_blocks} ブロック")
        if args.verbose:
            for rel, count in sorted(result.items()):
                out = base_dir / rel
                size = out.stat().st_size
                print(f"  {rel}  ({count} ブロック, {size:,} bytes)")
        else:
            for rel in sorted(result.keys()):
                print(f"  {rel}")


if __name__ == "__main__":
    main()

"""
RBAC モジュール
本番はKeycloak JWT検証に差し替える。
現在はヘッダー X-User-Name / X-User-Role によるシンプル認証。
X-User-Name は btoa(encodeURIComponent(name)) でBase64エンコードされている。
"""
from fastapi import Header, HTTPException, Depends
from sqlalchemy.orm import Session
from sqlalchemy import text
from database import get_db
from typing import Optional
import base64
from urllib.parse import unquote

ROLE_PERMISSIONS = {
    "admin":     {"read", "write", "confirm", "export", "manage"},
    "evaluator": {"read", "write", "confirm"},
    "pjmo":      {"read", "write"},
    "viewer":    {"read"},
}


def _decode_username(raw: str) -> str:
    """btoa(encodeURIComponent(name)) → 元の文字列に戻す"""
    try:
        # Base64デコード → URLデコード
        decoded = base64.b64decode(raw + "==").decode("utf-8")
        return unquote(decoded)
    except Exception:
        return raw  # デコード失敗時はそのまま使用


class UserContext:
    def __init__(self, username: str, display_name: str, role: str):
        self.username = username
        self.display_name = display_name
        self.role = role

    def can(self, permission: str) -> bool:
        return permission in ROLE_PERMISSIONS.get(self.role, set())

    def require(self, permission: str):
        if not self.can(permission):
            raise HTTPException(
                status_code=403,
                detail=f"権限不足: {permission} が必要です（現在のロール: {self.role}）"
            )


def get_current_user(
    x_user_name: Optional[str] = Header(default=None),
    x_user_role: Optional[str] = Header(default=None),
    db: Session = Depends(get_db),
) -> UserContext:
    if not x_user_name:
        return UserContext("anonymous", "匿名ユーザー", "viewer")

    username = _decode_username(x_user_name)

    row = db.execute(
        text("SELECT display_name, role FROM users WHERE username=:u"),
        {"u": username}
    ).fetchone()

    if row:
        return UserContext(username, row.display_name, row.role)

    valid_roles = set(ROLE_PERMISSIONS.keys())
    role = x_user_role if x_user_role in valid_roles else "viewer"
    return UserContext(username, username, role)

def build_agent_card(event_type: str, log: dict, project_name: str) -> dict:
    """
    A2A互換 Agent Card 形式で通知ペイロードを生成。
    受信側が MCP エージェントでも汎用 REST でも処理できる形式。
    """
    STATUS_LABEL = {"issue": "指摘", "reply": "回答", "review": "再評価", "close": "確定"}

            "name": log["author"],
            "jobTitle": log["role"]
        },
        "object": {
            "@type": "GovernmentService",
            "name": project_name,
        },
        "actionStatus": event_type,
        "startTime": log.get("created_at", ""),
        "pjmo_metadata": {
            "project_id": log["project_id"],
            "log_id": log["id"],
            "status": log["status"],
            "version": log.get("version", 1),
        }
    }


async def _send_webhook(payload: dict, project_id: int, event_type: str, db: Session):
    """バックグラウンドでWebhook送信し webhook_logs に記録"""
    if not WEBHOOK_URL:
        return

    sent_at = None
    status = "failed"
    try:
        async with httpx.AsyncClient(timeout=10) as client:
            resp = await client.post(
                WEBHOOK_URL,
                json=payload,
                headers={"Content-Type": "application/json"}
            )
        status = "sent" if resp.status_code < 400 else "failed"
        sent_at = datetime.now()
    except Exception as e:
        status = f"error: {str(e)[:100]}"

    db.execute(
        text("""
            INSERT INTO webhook_logs (project_id, event_type, payload, status, sent_at)
            VALUES (:p, :e, CAST(:payload AS jsonb), :s, :t)
        """),
        {
            "p": project_id,
            "e": event_type,
            "payload": json.dumps(payload, ensure_ascii=False),
            "s": status,
            "t": sent_at,
        }
    )
    db.commit()

def _export_system(db, system_id: int) -> dict:
    sys_row = db.execute(
        text("SELECT id, name, department FROM systems WHERE id=:id"),
        {"id": system_id}
    ).fetchone()
    if not sys_row:
        raise HTTPException(404, "システムが見つかりません")

    projects = db.execute(text("""
        SELECT id, name, cost_category, is_system_wide,
               amount_single, amount_advance, amount_deferred, evaluator_name,
               estimate_text, requirements_text, basis_text
        FROM projects WHERE system_id=:s ORDER BY is_system_wide DESC, id
    """), {"s": system_id}).fetchall()

    result_projects = []
    for p in projects:
        threads = db.execute(text("""
            SELECT id, thread_no, title, status, due_date::text
            FROM review_threads WHERE project_id=:p ORDER BY thread_no
        """), {"p": p.id}).fetchall()

        result_threads = []
        for t in threads:
            entries = db.execute(text("""
                SELECT seq, entry_type, role, comment, entry_date::text
                FROM review_entries WHERE thread_id=:t ORDER BY seq
            """), {"t": t.id}).fetchall()
            result_threads.append({
                "thread_no": t.thread_no,
                "title": t.title,
                "status": t.status,
                "due_date": t.due_date,
                "entries": [dict(e._mapping) for e in entries]
            })

        result_projects.append({
            "name": p.name,
            "is_system_wide": p.is_system_wide,
            "cost_category": p.cost_category,
            "amount_single": p.amount_single,
            "amount_advance": p.amount_advance,
            "amount_deferred": p.amount_deferred,
            "evaluator_name": p.evaluator_name,
            "estimate_text": p.estimate_text,
            "requirements_text": p.requirements_text,
            "basis_text": p.basis_text,
            "threads": result_threads,
        })

    return {
        "export_version": "1.0",
        "exported_at": datetime.now().isoformat(),
        "system": {"id": sys_row.id, "name": sys_row.name, "department": sys_row.department},
        "projects": result_projects,
    }

@router.post("/import/project/{project_id}/relay")
def import_project_relay(project_id: int, payload: dict,
                         mode: str = "append",
                         db: Session = Depends(get_db),
                         user: UserContext = Depends(get_current_user)):
    """
    「レビューラリー」履歴のインポート。
    - mode='append'  : 既存スレッドを残し、thread_no を採番し直して追記
    - mode='replace' : 事業配下の threads/entries を全削除してから投入
    """
    user.require("manage")
    if payload.get("export_version") != "1.0":
        raise HTTPException(400, "サポートされていない export_version です")
    if payload.get("export_type") != "relay":
        raise HTTPException(400, "リレーデータの形式ではありません")
    if mode not in ("append", "replace"):
        raise HTTPException(400, "mode は append または replace を指定してください")

    pj = db.execute(
        text("SELECT id FROM projects WHERE id=:id"), {"id": project_id}
    ).fetchone()
    if not pj:
        raise HTTPException(404, "事業が見つかりません")

    if mode == "replace":
        db.execute(text("""
            DELETE FROM review_entries
            WHERE thread_id IN (SELECT id FROM review_threads WHERE project_id=:p)
        """), {"p": project_id})
        db.execute(text("DELETE FROM review_threads WHERE project_id=:p"),
                   {"p": project_id})

    # 既存の最大 thread_no を取得（append時の採番用）
    max_no = db.execute(
        text("SELECT COALESCE(MAX(thread_no),0) FROM review_threads WHERE project_id=:p"),
        {"p": project_id}
    ).scalar() or 0

    imported_threads = 0
    imported_entries = 0
    for i, t in enumerate(payload.get("threads", []), start=1):
        thread_no = (max_no + i) if mode == "append" else t.get("thread_no", i)
        t_row = db.execute(text("""
            INSERT INTO review_threads (project_id, thread_no, title, status, due_date)
            VALUES (:p,:tn,:ti,:st,:dd) RETURNING id
        """), {"p": project_id, "tn": thread_no, "ti": t.get("title"),
               "st": t.get("status", "open"), "dd": t.get("due_date")}).fetchone()
        thread_id = t_row[0]
        imported_threads += 1

        for e in t.get("entries", []):
            db.execute(text("""
                INSERT INTO review_entries (thread_id, seq, entry_type, role, comment, entry_date)
                VALUES (:t,:s,:et,:r,:c,:ed)
            """), {"t": thread_id, "s": e.get("seq"), "et": e.get("entry_type"),
                   "r": e.get("role"), "c": e.get("comment", ""),
                   "ed": e.get("entry_date")})
            imported_entries += 1

    db.commit()
    return {
        "status": "ok",
        "mode": mode,
        "project_id": project_id,
        "imported_threads": imported_threads,
        "imported_entries": imported_entries,
    }

    setLoading(true);

      const r = await fetch(`${api}/advisor/chat`, {
        method: "POST",
        headers: { "Content-Type": "application/json", ...headers() },

          messages: newMessages.map(m => ({ role: m.role, content: m.content })),
          system: SYSTEM_PROMPT,
        }),
      });
      const raw = await r.text();

        let msg = `HTTP ${r.status}`;
        try { msg = JSON.parse(raw)?.detail ?? msg; } catch {}
        throw new Error(msg);

      const data = JSON.parse(raw);

function SystemsView({ headers, showToast }: { headers: () => Record<string, string>; showToast: (m: string) => void }) {
  const [systems, setSystems] = useState<SystemRow[]>([]);
  const [projects, setProjects] = useState<Record<number, ProjectRow[]>>({});
  const [expandedSys, setExpandedSys] = useState<number | null>(null);
  const [showNewSys, setShowNewSys] = useState(false);
  const [showNewPj, setShowNewPj] = useState<number | null>(null);
  const [newSysName, setNewSysName] = useState("");
  const [newSysDept, setNewSysDept] = useState("");
  const [newPj, setNewPj] = useState({ name: "", cost_category: "", amount_single: "", amount_advance: "", amount_deferred: "", evaluator_name: "" });

  // 削除確認モーダル
  const [confirmDelete, setConfirmDelete] = useState<SystemRow | null>(null);
  // インポートモーダル
  const [showImport, setShowImport] = useState(false);
  const importFileRef = useRef<HTMLInputElement>(null);

  const loadSystems = useCallback(() => {
    fetch(`${API}/systems/`, { headers: headers() }).then(r => r.json()).then(setSystems);
  }, []);

  const loadProjects = (sysId: number) => {
    fetch(`${API}/systems/${sysId}/projects`, { headers: headers() })
      .then(r => r.json()).then(data => setProjects(p => ({ ...p, [sysId]: data })));
  };

  useEffect(() => { loadSystems(); }, [loadSystems]);

  const createSystem = () => {
    if (!newSysName.trim()) return;
    fetch(`${API}/systems/`, { method: "POST",
      headers: { "Content-Type": "application/json", ...headers() },
      body: JSON.stringify({ name: newSysName, department: newSysDept })
    }).then(() => { setNewSysName(""); setNewSysDept(""); setShowNewSys(false); loadSystems(); showToast("システムを追加しました"); });
  };

  const createProject = (sysId: number) => {
    if (!newPj.name.trim()) return;
    fetch(`${API}/projects/`, { method: "POST",
      headers: { "Content-Type": "application/json", ...headers() },
      body: JSON.stringify({ system_id: sysId, name: newPj.name,
        cost_category: newPj.cost_category || null,
        amount_single: Number(newPj.amount_single) || 0,
        amount_advance: Number(newPj.amount_advance) || 0,
        amount_deferred: Number(newPj.amount_deferred) || 0,
        evaluator_name: newPj.evaluator_name || null })
    }).then(() => {
      setNewPj({ name: "", cost_category: "", amount_single: "", amount_advance: "", amount_deferred: "", evaluator_name: "" });
      setShowNewPj(null); loadProjects(sysId); loadSystems(); showToast("事業を追加しました");
    });
  };

  // システム削除
  const deleteSystem = (sys: SystemRow) => {
    fetch(`${API}/systems/${sys.id}`, { method: "DELETE", headers: headers() })
      .then(async r => {
        if (!r.ok) {
          const err = await r.json().catch(() => ({ detail: "削除に失敗しました" }));
          throw new Error(err.detail || "削除に失敗しました");
        }
        setConfirmDelete(null);
        if (expandedSys === sys.id) setExpandedSys(null);
        loadSystems();
        showToast(`「${sys.name}」を削除しました`);
      })
      .catch((e: Error) => showToast(`削除エラー: ${e.message}`));
  };

  // システムのエクスポート（JSONダウンロード）
  const exportSystem = (sys: SystemRow) => {
    fetch(`${API}/export/system/${sys.id}/json`, { headers: headers() })
      .then(async r => {
        if (!r.ok) throw new Error("エクスポートに失敗しました");
        return r.json();
      })
      .then(data => {
        const blob = new Blob([JSON.stringify(data, null, 2)], { type: "application/json" });
        const url  = URL.createObjectURL(blob);
        const a    = document.createElement("a");
        a.href = url;
        a.download = `system_${sys.id}_${sys.name}_${new Date().toISOString().slice(0,10)}.json`;
        document.body.appendChild(a); a.click(); document.body.removeChild(a);
        URL.revokeObjectURL(url);
        showToast(`「${sys.name}」をエクスポートしました`);
      })
      .catch((e: Error) => showToast(`エクスポートエラー: ${e.message}`));
  };

  // システムのインポート（JSONファイル選択）
  const importSystem = (file: File) => {
    const reader = new FileReader();
    reader.onload = () => {
      try {
        const payload = JSON.parse(String(reader.result));
        fetch(`${API}/import/system`, { method: "POST",
          headers: { "Content-Type": "application/json", ...headers() },
          body: JSON.stringify(payload)
        }).then(async r => {
          if (!r.ok) {
            const err = await r.json().catch(() => ({ detail: "インポートに失敗しました" }));
            throw new Error(err.detail || "インポートに失敗しました");
          }
          setShowImport(false);
          loadSystems();
          showToast("システムをインポートしました");
        }).catch((e: Error) => showToast(`インポートエラー: ${e.message}`));
      } catch {
        showToast("JSONの解析に失敗しました");
      }
    };
    reader.readAsText(file);
  };

  const fmtAmount = (n: number) => n > 0 ? `${n.toLocaleString()} 千円` : "—";

  return (
    <div>
      <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center", marginBottom: 16 }}>
        <h2 style={{ ...S.h2, margin: 0 }}>システム管理</h2>
        <div style={{ display: "flex", gap: 8 }}>
          <button onClick={() => setShowImport(true)} style={chipBtn()}>📥 インポート</button>
          <button onClick={() => setShowNewSys(true)} style={btn("#4338ca")}>＋ システム追加</button>
        </div>
      </div>

      {/* 新規システムモーダル */}
      {showNewSys && (
        <div style={{ position: "fixed", inset: 0, background: "rgba(0,0,0,.4)", zIndex: 1000,
                      display: "flex", alignItems: "center", justifyContent: "center" }}
             onClick={() => setShowNewSys(false)}>
          <div style={{ ...S.card, width: 400 }} onClick={e => e.stopPropagation()}>
            <h3 style={{ margin: "0 0 16px", fontSize: 15 }}>評価対象システムを追加</h3>
            <div style={{ marginBottom: 12 }}>
              <label style={{ fontSize: 11, color: "#94a3b8", display: "block", marginBottom: 4 }}>システム名 *</label>
              <input value={newSysName} onChange={e => setNewSysName(e.target.value)}
                placeholder="例：人事給与システム（次期）"
                style={{ ...S.input, width: "100%", boxSizing: "border-box" as const }} />
            </div>
            <div style={{ marginBottom: 20 }}>
              <label style={{ fontSize: 11, color: "#94a3b8", display: "block", marginBottom: 4 }}>担当部署</label>
              <input value={newSysDept} onChange={e => setNewSysDept(e.target.value)}
                placeholder="例：人事院"
                style={{ ...S.input, width: "100%", boxSizing: "border-box" as const }} />
            </div>
            <div style={{ display: "flex", gap: 8, justifyContent: "flex-end" }}>
              <button onClick={() => setShowNewSys(false)} style={chipBtn()}>キャンセル</button>
              <button onClick={createSystem} disabled={!newSysName.trim()} style={btn("#0f172a", !newSysName.trim())}>追加</button>
            </div>
          </div>
        </div>
      )}

      {/* 削除確認モーダル */}
      {confirmDelete && (

** nginx/gen-cert.sh

#+begin_src shell :tangle nginx/gen-cert.sh :shebang "#!/bin/bash"
set -e
CERT_DIR="$(dirname "$0")/certs"
mkdir -p "$CERT_DIR"
if [ -f "$CERT_DIR/server.crt" ]; then
  echo "証明書が既に存在します: $CERT_DIR/server.crt"
  exit 0
fi
openssl req -x509 -nodes -days 365 -newkey rsa:2048 \
  -keyout "$CERT_DIR/server.key" \
  -out "$CERT_DIR/server.crt" \
  -subj "/C=JP/ST=Tokyo/L=Chiyoda/O=PJMO/CN=pjmo-review.local" \
  -addext "subjectAltName=DNS:pjmo-review.local,DNS:localhost,IP:127.0.0.1"
chmod 600 "$CERT_DIR/server.key"
echo "完了: $CERT_DIR/"
#+end_src

* トラブルシューティング履歴

** DBテーブル未作成エラー（解決済み）

=psycopg2.errors.UndefinedTable: relation "systems" does not exist= が
運用中に発生。原因は次の2点の複合だった。

*** 原因1: init.sql の二重 CREATE TABLE 定義

=init.sql= 内で =systems= / =projects= / =review_threads= の
=CREATE TABLE IF NOT EXISTS= が *2ブロック* 書かれていた。
2ブロック目は =projects= に =name= 列を持たないなど整合しない壊れた簡略定義で、
=IF NOT EXISTS= のため実行時には無視されるが、他の開発者がそちらに合わせて
誤修正を入れる温床になっていたため、完全に削除した。

*** 原因2: 古い pgdata ボリュームの残存（*これが直接原因*）

PostgreSQL 公式イメージの =/docker-entrypoint-initdb.d/= は
*データディレクトリが空のとき* しか実行されない。
以前の壊れた =init.sql= で初期化された =pgdata= が残っていたため、
修正版 =init.sql= は再実行されず、DBに =systems= テーブルが存在しないまま
API が叩かれて500エラーとなっていた。

*** 対処手順

#+begin_src shell :eval no
# 開発環境（DB内データを捨てて良い場合）
docker compose down -v          # -v で pgdata ボリュームも削除
docker compose build --no-cache
docker compose up

# 起動後にテーブル作成を確認
docker compose exec db psql -U budget -d budget -c "\dt"
# 期待: systems, projects, review_threads, review_entries, users,
#       project_files, webhook_logs, audit_logs が表示されること
#+end_src

#+begin_src shell :eval no
# 本番・保持したいデータがある場合は init.sql を手動再適用
# （CREATE TABLE IF NOT EXISTS なので既存テーブルは保持される）
docker compose exec -T db psql -U budget -d budget < db/init.sql
#+end_src

| エラー                                              | 原因                                 | 対処                          |
|-----------------------------------------------------+--------------------------------------+-------------------------------|
| =relation "systems" does not exist=                 | 旧 pgdata 残存で init.sql 未再実行   | =docker compose down -v= で再初期化 |
| =init.sql= に二重の CREATE TABLE                    | 過去コミットの残骸                   | 重複ブロックを削除            |

** フロントエンドビルド問題（解決済み）

=react-scripts 5 + Node 18/20= の組み合わせで発生した =ajv= / =fork-ts-checker-webpack-plugin= の依存関係競合を、*Vite 5 + Node 20* への移行で根本解決した。

| エラー | 原因 | 対処 |
|--------+------+------|
| =ajv/dist/compile/codegen= not found | fork-ts-checker が ajv@6 を内包 | Vite 移行 |
| =Unknown keyword formatMinimum= | ajv-keywords バージョン不整合 | Vite 移行 |
| =schema_utils_1.default is not a function= | schema-utils バージョン不整合 | Vite 移行 |
| =port is already allocated= | ホスト側3000番使用中 | ホスト側3005番に変更 |
| =version obsolete= | docker-compose.yml の version フィールド | 削除 |
